 52 void PmergeMe::createJacobsthalVector(std::vector<unsigned int>& vec, const size_t max_num) {
 53   unsigned int j_prev = 0, j_curr = 1;
 54 
 55   vec.push_back(j_prev);
 56 
 57   if (max_num == 0) {
 58     return;
 59   }
 60 
 61   while (true) {
 62     vec.push_back(j_curr);
 63     if (j_curr >= max_num) {
 64       break;
 65     }
 66     unsigned int j_next = j_curr + 2 * j_prev;
 67     j_prev = j_curr;
 68     j_curr = j_next;
 69   }
 70   if (vec.back() > max_num) {
 71     vec.pop_back();
 72     vec.push_back(max_num);
 73   }
 74 }

122 // vectorからPairを作成し、vector_chainを更新
123 // 大きい数字の下に小さい数字を入れ、vector_chainから小さい数字を削除する
124 std::vector<p_vec> PmergeMe::makePair(std::vector<p_vec>& vec){
125     for ( size_t i = 1; i < vec.size(); i++){
126       if (vec[i-1].num>=vec[i].num){
127         vec[i-1].p_vecs.push_back(vec[i]);
128         vec.erase(vec.begin() + i);
129       }
130       else if (vec[i-1].num<vec[i].num){
131         vec[i].p_vecs.push_back(vec[i-1]);
132         vec.erase(vec.begin() + i - 1);
133       }
134     }
135     return vec;
136 
137 }

155         return result;
156     }
157 
158     std::vector<int> jacobsthal;

160     jacobsthal.push_back(3);
161     int next = 3;
162 
163     while (jacobsthal.size() < static_cast<size_t>(n)) {
164         next = jacobsthal[jacobsthal.size() - 1] + 2 * jacobsthal[jacobsthal.size() - 2];
165         if (next > n) break;
166         jacobsthal.push_back(next);

168 
169     std::vector<int> result;
170     result.push_back(1);
171 
172     for (size_t i = 1; i < jacobsthal.size(); ++i) {
173         for (int j = jacobsthal[i]; j > jacobsthal[i-1]; --j) {
174             if (j <= n) result.push_back(j);

176     }

181     std::cout << "resul[i] " << result[i] << std::endl;

183       if ( result[i] > n){
